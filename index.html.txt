  <script>
    const ipValue = document.getElementById('ipValue');
    const ipNote = document.getElementById('ipNote');
    const extraInfo = document.getElementById('extraInfo');
    const refreshBtn = document.getElementById('refreshBtn');
    const copyBtn = document.getElementById('copyBtn');
    const copyStatus = document.getElementById('copyStatus');

    async function fetchIP(){
      ipValue.textContent = '—';
      ipNote.textContent = 'Fetching your public IP…';
      extraInfo.textContent = 'loading…';
      try {
        const res = await fetch('https://api.ipify.org?format=json');
        if(!res.ok) throw new Error('API error: ' + res.status);
        const data = await res.json();
        const ip = data.ip || 'unknown';
        ipValue.textContent = ip;
        ipNote.textContent = 'This is your public IP address (what websites see).';
        const ua = navigator.userAgent;
        extraInfo.textContent = JSON.stringify({
          ip,
          userAgent: ua,
          language: navigator.language || '',
          platform: navigator.platform || '',
          timestamp: new Date().toISOString()
        }, null, 2);
      } catch(err){
        ipValue.textContent = 'Unavailable';
        ipNote.textContent = 'Could not fetch IP — API blocked or offline.';
        extraInfo.textContent = String(err);
      }
    }

    refreshBtn.addEventListener('click', () => {
      fetchIP();
      copyStatus.textContent = '';
    });

    copyBtn.addEventListener('click', async () => {
      const text = ipValue.textContent;
      if(!text || text === '—' || text === 'Unavailable') {
        copyStatus.textContent = 'Nothing to copy';
        return;
      }
      try {
        await navigator.clipboard.writeText(text);
        copyStatus.textContent = 'Copied!';
      } catch(e){
        copyStatus.textContent = 'Copy failed — try selecting manually.';
      }
    });

    fetchIP();
  </script>
